Required patterns from RESEARCH.md:

Safe removal with tracking arrays:
```bash
declare -a REMOVED=()
declare -a SKIPPED=()

safe_rm_dir() {
    local dir="$1"
    local description="$2"
    if [[ -d "$dir" ]]; then
        rm -rf "$dir"
        REMOVED+=("$description: $dir")
    else
        SKIPPED+=("$description: $dir (not found)")
    fi
}

safe_rm_file() {
    local file="$1"
    local description="$2"
    if [[ -f "$file" ]]; then
        rm -f "$file"
        REMOVED+=("$description: $file")
    else
        SKIPPED+=("$description: $file (not found)")
    fi
}

safe_make_uninstall() {
    local dir="$1"
    local component="$2"
    if [[ -d "$dir" ]]; then
        cd "$dir"
        if make uninstall 2>/dev/null; then
            REMOVED+=("make uninstall: $component")
        else
            SKIPPED+=("make uninstall: $component (failed or no target)")
        fi
    else
        SKIPPED+=("build directory: $dir (not found)")
    fi
}
```

